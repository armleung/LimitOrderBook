cmake_minimum_required(VERSION 3.10)

# Set the project name
project(OrderBookDemo)

# Specify the C++ standard
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED True)

# Add the executable
add_executable(OrderBookDemo
    src/main.cpp
    src/OrderBook.cpp
)

# Include directories
target_include_directories(OrderBookDemo PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/src)
